int n, k, i, count;
typedef int bool;
int primes[100];

input(k);

count = 0;

for (n = 2; count < k; n = n + 1) {
    bool is_prime = 1;

    for (i = 2; i < n; i = i + 1) {
        if (n % i == 0) {
            is_prime = 0;
            break;
        }
    }

    if (is_prime == 1) {
        primes[count] = n;
        count = count + 1;
    }
}

output(primes);
